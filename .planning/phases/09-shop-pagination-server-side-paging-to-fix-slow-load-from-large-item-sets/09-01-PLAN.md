<objective>
Introduce server-side pagination + server-side name/description search at the data layer for the player-facing shop.

Purpose: Today `ShopController.Index` loads the entire published item list into memory and applies rarity filter + sort post-fetch. This plan pushes filtering, sorting, search, and pagination into one EF Core query at the repository layer, and grows the domain/service contract + ViewModel to support it. No controller/view changes happen here — Plan 02 handles those.

Output: A unified paged repository method, matching service method, extended ViewModel, and test coverage proving the new contract.
</objective>

       ```csharp
       public async Task<(IList<ShopItem> Items, int TotalCount)> GetPagedPublishedItemsAsync(
           ItemType? type,
           IList<ItemRarity>? rarities,
           string? sort,
           string? search,
           int page,
           int pageSize,
           CancellationToken token = default)
       {
           var query = DbContext.ShopItems
               .Include(si => si.CreatedByDm)
               .Where(si => si.Status == 1)
               .Where(si => si.AvailableFrom == null || si.AvailableFrom <= DateTime.UtcNow)
               .Where(si => si.AvailableUntil == null || si.AvailableUntil >= DateTime.UtcNow);

           if (type.HasValue)
               query = query.Where(si => si.Type == (int)type.Value);

           if (rarities is { Count: > 0 })
           {
               var rarityInts = rarities.Select(r => (int)r).ToList();
               query = query.Where(si => rarityInts.Contains(si.Rarity));
           }

           if (!string.IsNullOrWhiteSpace(search))
               query = query.Where(si => si.Name.Contains(search) || si.Description.Contains(search));

           var totalCount = await query.CountAsync(cancellationToken: token);

           query = sort switch
           {
               "price_asc" => query.OrderBy(si => si.Price),
               "price_desc" => query.OrderByDescending(si => si.Price),
               _ => query.OrderBy(si => si.Name)
           };

           var entities = await query
               .Skip((page - 1) * pageSize)
               .Take(pageSize)
               .ToListAsync(cancellationToken: token);

           return (Mapper.Map<IList<ShopItem>>(entities), totalCount);
       }
       ```
