```cshtml
@using EuphoriaInn.Service.ViewModels.DungeonMasterViewModels
@model DMProfileViewModel
@{
    ViewData["Title"] = $"{Model.Name} — DM Profile";
}

@section Styles {
    <link href="~/css/dm-profile.mobile.css" asp-append-version="true" rel="stylesheet" />
}

<div class="container-fluid px-2 mt-2">

    @* Header card: photo + name + edit button *@
    <div class="dm-profile-header-card mb-3 text-center">
        @if (Model.HasProfilePicture)
        {
            <img src="@Url.Action("GetDMProfilePicture", new { id = Model.UserId })"
                 alt="@Model.Name"
                 class="rounded-circle dm-profile-photo-mobile mb-3" />
        }
        else
        {
            <div class="dm-profile-placeholder-mobile mx-auto mb-3">
                <i class="fas fa-user fa-4x" style="color: rgba(244,228,188,0.5);"></i>
            </div>
        }
        <h5 class="dm-profile-name mb-1">@Model.Name</h5>
        <p class="dm-profile-role-label">
            <i class="fas fa-crown me-1"></i>Dungeon Master
        </p>
        @if (Model.CanEdit)
        {
            <a asp-action="EditProfile" asp-route-id="@Model.UserId"
               class="btn btn-warning w-100 mt-2">
                <i class="fas fa-edit me-2"></i>Edit Profile
            </a>
        }
    </div>

    @* Bio card *@
    <div class="dm-profile-bio-card mb-3">
        <h6 class="dm-profile-section-heading mb-2">
            <i class="fas fa-scroll me-2"></i>About
        </h6>
        @if (!string.IsNullOrWhiteSpace(Model.Bio))
        {
            <p class="dm-profile-bio-text">@Model.Bio</p>
        }
        else
        {
            <p class="dm-profile-bio-empty">No bio provided yet.</p>
        }
    </div>

    @* Quest history card *@
    <div class="dm-profile-history-card mb-3">
        <h6 class="dm-profile-section-heading mb-2">
            <i class="fas fa-list me-2"></i>Quest History
        </h6>
        @if (Model.Quests.Any())
        {
            @foreach (var quest in Model.Quests)
            {
                <div class="dm-quest-history-item"
                     onclick="window.location.href='@Url.Action("Details", "Quest", new { id = quest.Id })'">
                    <div class="d-flex justify-content-between align-items-start">
                        <span class="dm-quest-history-title fw-bold">@quest.Title</span>
                        <span class="badge bg-secondary ms-2 flex-shrink-0">
                            CR @quest.ChallengeRating
                        </span>
                    </div>
                    <small class="dm-quest-history-date d-block mt-1">
                        <i class="fas fa-calendar-check me-1"></i>
                        @(quest.Date?.ToString("MMM dd, yyyy") ?? "Date TBD")
                    </small>
                </div>
            }
        }
        else
        {
            <div class="text-center py-4">
                <i class="fas fa-dragon fa-3x text-warning mb-3"></i>
                <h5 class="dm-profile-name">No Quests Yet</h5>
                <p class="dm-profile-bio-empty">This Dungeon Master hasn't run any quests yet.</p>
            </div>
        }
    </div>

</div>
```

Key implementation notes:
- Three separate glass cards (D-08): `dm-profile-header-card`, `dm-profile-bio-card`, `dm-profile-history-card`.

- Empty state: `fa-dragon fa-3x text-warning`, "No Quests Yet" heading in `dm-profile-name` style, body in `dm-profile-bio-empty` style.
- No @section Scripts — Profile.cshtml has none; mobile variant inherits this.
