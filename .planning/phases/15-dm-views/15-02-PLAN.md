@using EuphoriaInn.Domain.Interfaces
@using EuphoriaInn.Service.ViewModels.QuestViewModels
@model QuestViewModel
@{

}

@section Styles {
    <link href="~/css/dm-create.mobile.css" asp-append-version="true" rel="stylesheet" />
}

<div class="container-fluid px-2 mt-2">
    <div class="dm-create-card-mobile mb-3">

        <div class="mb-3">
            <h5 class="mb-0">
                <i class="fas fa-scroll text-warning me-2"></i>Create New Quest
            </h5>
        </div>

        <div asp-validation-summary="All" class="text-danger mb-3"></div>

        <form asp-action="Create" method="post">

            <div class="mb-3">
                <label asp-for="Title" class="form-label dm-create-label">
                    Quest Title <span class="text-danger">*</span>
                </label>
                <input asp-for="Title" class="form-control" placeholder="Enter the quest title..." />
                <span asp-validation-for="Title" class="text-danger"></span>
            </div>

            <div class="mb-3">
                <label asp-for="Description" class="form-label dm-create-label">
                    Description <span class="text-danger">*</span>
                </label>
                <textarea asp-for="Description" class="form-control" rows="4"
                          placeholder="Describe the quest..."></textarea>
                <span asp-validation-for="Description" class="text-danger"></span>
            </div>

            <div class="mb-3">
                <label asp-for="ChallengeRating" class="form-label dm-create-label">
                    Challenge Rating <span class="text-danger">*</span>
                </label>
                <input asp-for="ChallengeRating" type="number" class="form-control"
                       min="1" max="20" step="1" placeholder="Enter level (e.g., 1, 5, 10, 15)" />
                <span asp-validation-for="ChallengeRating" class="text-danger"></span>
            </div>

            <div class="mb-3">
                <label asp-for="TotalPlayerCount" class="form-label dm-create-label">
                    Total Player Count <span class="text-danger">*</span>
                </label>
                <input asp-for="TotalPlayerCount" inputmode="numeric" class="form-control" />
                <span asp-validation-for="TotalPlayerCount" class="text-danger"></span>
            </div>

            <div class="mb-3">
                <div class="form-check">
                    <input asp-for="DungeonMasterSession" class="form-check-input" type="checkbox" />
                    <label asp-for="DungeonMasterSession" class="form-check-label">
                        <i class="fas fa-crown text-warning me-2"></i>Dungeon Master Session Only
                    </label>
                </div>
                <div class="form-text">Check to make this quest visible to DMs and Admins only.</div>
            </div>

            <div class="mb-3">
                <label class="form-label dm-create-label">
                    Proposed Dates &amp; Times <span class="text-danger">*</span>
                </label>
                <div id="proposed-dates">
                    @for (int i = 0; i < Model.ProposedDates.Count; i++)
                    {
                        <div class="mb-3 proposed-date-item">
                            <label class="form-label small">Proposed Date @(i + 1)</label>
                            <div class="input-group">
                                <input asp-for="ProposedDates[i]" type="datetime-local"
                                       class="form-control" required step="60" />

                                        onclick="removeProposedDate(this)">
                                    <i class="fas fa-trash me-1"></i>Remove

                            </div>

                </div>
                <button type="button" class="btn btn-warning btn-sm" onclick="addProposedDate()">
                    <i class="fas fa-plus me-1"></i>Add Another Date Option

    </div>
</div>

@section Scripts {
    @{ await Html.RenderPartialAsync("_QuestFormScripts"); }
}

Key implementation notes:
- The `Proposed Dates & Times` label uses `&amp;` HTML entity for the ampersand (Razor HTML-encodes raw `&` in attribute contexts — use `&amp;` in static text or `@:` to emit literal text).
- `rows="4"` on the Description textarea gives adequate mobile height.
- No sidebar (Tips card omitted per D-01).
- `input-group` layout retained on date rows per D-02.
- Both action buttons use `flex-fill` to share equal width per UI-SPEC form action row.
- `@section Scripts` declaration is in addition to `@section Styles` — both sections are valid in the same view; `_Layout.Mobile.cshtml` renders both.
