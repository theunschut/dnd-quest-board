                const errorDiv = document.getElementById('fileSizeError');
                errorDiv.style.display = 'none';
                errorDiv.textContent = '';
                if (file) {
                    if (!ALLOWED_TYPES.includes(file.type)) {
                        errorDiv.textContent = 'Only image files (JPG, PNG, GIF) are allowed.';
                        errorDiv.style.display = 'block';
                        e.target.value = '';
                        return;
                    }
                    if (file.size > MAX_FILE_SIZE) {
                        const sizeMB = (file.size / (1024 * 1024)).toFixed(2);
                        errorDiv.textContent = `File size (${sizeMB} MB) exceeds the maximum allowed size of 5 MB. Please choose a smaller image.`;

## Trust Boundaries

| Boundary | Description |

| browser→server | POST /GuildMembers/Create and /GuildMembers/Edit — authenticated ([Authorize]); file upload via multipart/form-data |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-17-08 | Tampering | Create/Edit POST antiforgery | mitigate | `asp-action` tag helper injects `__RequestVerificationToken`; `[ValidateAntiForgeryToken]` on controller POST actions (existing desktop validation unchanged — mobile view inherits same protection) |
