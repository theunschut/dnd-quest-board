```cshtml
@using EuphoriaInn.Domain.Interfaces
@using EuphoriaInn.Service.ViewModels.CalendarViewModels
@model CalendarViewModel
@{
    ViewData["Title"] = "Quest Calendar";
    ViewBag.IsDetailsPage = false;
    var prevMonth = Model.FirstDayOfMonth.AddMonths(-1);
    var nextMonth = Model.FirstDayOfMonth.AddMonths(1);
    var agendaDays = Model.GetCalendarDays().Where(d => !d.IsEmpty && d.QuestsOnDay.Any()).ToList();
}
@section Styles {
    <link href="~/css/calendar.mobile.css" asp-append-version="true" rel="stylesheet" />
}

<div class="container-fluid px-2 mt-2">

    @* Month navigation bar (D-01, D-02) *@
    <div class="d-flex justify-content-between align-items-center mb-3 calendar-month-nav">
        <a href="@Url.Action("Index", new { year = prevMonth.Year, month = prevMonth.Month })"
           class="btn btn-primary calendar-nav-btn" aria-label="Previous month">
            <i class="fas fa-chevron-left"></i>
        </a>
        <span class="fw-bold cinzel-heading calendar-month-title">
            @Model.MonthName
        </span>
        <a href="@Url.Action("Index", new { year = nextMonth.Year, month = nextMonth.Month })"
           class="btn btn-primary calendar-nav-btn" aria-label="Next month">
            <i class="fas fa-chevron-right"></i>
        </a>
    </div>

    @* Agenda list — one section per day that has at least one quest (CAL-01, CAL-03) *@
    @if (!agendaDays.Any())
    {
        @* Empty state (month with no quests) *@
        <div class="text-center text-muted py-5">
            <i class="fas fa-calendar-alt fa-3x mb-3"></i>
            <h5>No Quests This Month</h5>
            <p>No adventures are planned for @Model.MonthName. Check another month.</p>
        </div>
    }
    else
    {
        @foreach (var day in agendaDays)
        {
            <div class="agenda-day-section mb-2">
                @* Day label: SATURDAY, JUNE 14 (CAL-02, D-10) *@
                <div class="agenda-day-label text-uppercase fw-bold small mb-1">
                    @day.Date.ToString("dddd, MMMM d").ToUpper()
                </div>
                @* Quest entries for this day — all quests, no Take(3) cap (D-09) *@
                @foreach (var questOnDay in day.QuestsOnDay)
                {
                    <div class="agenda-quest-entry @(questOnDay.IsFinalized ? "agenda-quest-finalized" : "agenda-quest-proposed")"
                         onclick="window.location.href='@Url.Action("Details", "Quest", new { id = questOnDay.Quest.Id })'">
                        <div class="d-flex justify-content-between align-items-center">
                            <span class="agenda-quest-title fw-bold">@questOnDay.Quest.Title</span>
                            <small class="text-muted ms-2 flex-shrink-0">
                                @questOnDay.ProposedDate.Date.ToString("HH:mm")
                            </small>
                        </div>
                    </div>
                }
            </div>
        }
    }

</div>
```

Key rules to follow (per CONTEXT.md and RESEARCH.md):
- Do NOT use `.Take(3)` — agenda shows ALL quests per day
- Do NOT add `@inject` — globally injected via _ViewImports.cshtml
- Do NOT add `@section Scripts` — this view has no JavaScript
- Do NOT add `@media` queries in CSS
- Day label uses `.ToUpper()` in C# (not CSS-only uppercase)
- Status class is determined by `questOnDay.IsFinalized` (bool, already computed by CalendarViewModel)
- Tap navigation uses `onclick="window.location.href='@Url.Action(...)'"`
- `ViewBag.IsDetailsPage = false` prevents the _Calendar partial from filtering to a specific quest (not used in this view, but defensively set)
