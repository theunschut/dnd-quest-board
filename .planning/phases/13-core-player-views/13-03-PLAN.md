```cshtml
@using EuphoriaInn.Domain.Enums
@using EuphoriaInn.Domain.Interfaces
@using EuphoriaInn.Domain.Models.QuestBoard
@using EuphoriaInn.Service.ViewModels.CalendarViewModels
@model PlayerSignup
@{
    ViewData["Title"] = Model.Quest?.Title;
    ViewData["BodyClass"] = "quest-details-page";
    var tokens = Antiforgery.GetAndStoreTokens(ViewContext.HttpContext);
    var currentUserId = ViewBag.CurrentUserId as int?;
    var isPlayerSignedUp = (bool)ViewBag.IsPlayerSignedUp;
    var canManage = (bool)ViewBag.CanManage;

    var selectedPlayers = Model.Quest?.PlayerSignups
        .Where(ps => ps.IsSelected && ps.Role == SignupRole.Player)
        .OrderBy(ps => ps.SignupTime).ToList() ?? [];
    var selectedAssistants = Model.Quest?.PlayerSignups
        .Where(ps => ps.IsSelected && ps.Role == SignupRole.AssistantDM)
        .OrderBy(ps => ps.SignupTime).ToList() ?? [];
    var selectedSpectators = Model.Quest?.PlayerSignups
        .Where(ps => ps.IsSelected && ps.Role == SignupRole.Spectator)
        .OrderBy(ps => ps.SignupTime).ToList() ?? [];
    var allSelectedParticipants = new List<PlayerSignup>();
    allSelectedParticipants.AddRange(selectedPlayers);
    allSelectedParticipants.AddRange(selectedAssistants);
    allSelectedParticipants.AddRange(selectedSpectators);
}
@section Styles {
    <link href="~/css/quests.mobile.css" asp-append-version="true" rel="stylesheet" />
}

<div class="container-fluid px-2 mt-2">

    @* Header: Title + CR *@
    <div class="d-flex justify-content-between align-items-start mb-2">
        <h5 class="mb-0 me-2">@Model.Quest?.Title</h5>
        <span class="badge bg-dark border flex-shrink-0">
            <i class="fas fa-dice-d20 me-1"></i>CR @Model.Quest?.ChallengeRating
        </span>
    </div>

    @* Meta: DM name + status *@
    @{
        string statusBadge;
        string statusText;
        if (Model.Quest?.IsFinalized == true && Model.Quest.FinalizedDate.HasValue
            && Model.Quest.FinalizedDate.Value.Date <= DateTime.UtcNow.AddDays(-1).Date)
        {
            statusBadge = "bg-secondary";
            statusText = "Done";
        }
        else if (Model.Quest?.IsFinalized == true)
        {
            statusBadge = "bg-primary";
            statusText = "Finalized";
        }
        else
        {
            statusBadge = "bg-success";
            statusText = "Open";
        }
    }
    <div class="d-flex justify-content-between align-items-center mb-3">
        <small class="text-muted">
            <i class="fas fa-crown me-1"></i>@Model.Quest?.DungeonMaster?.Name
        </small>
        <span class="badge @statusBadge">@statusText</span>
    </div>

    @* Description (above voting per Claude's discretion) *@
    @if (!string.IsNullOrWhiteSpace(Model.Quest?.Description))
    {
        <div class="quest-description-mobile mb-3">
            @Model.Quest.Description
        </div>
        <hr class="border-secondary mb-3" />
    }

    @* Voting section: update vote (signed-up player, open quest) *@
    @if (User.Identity?.IsAuthenticated == true && isPlayerSignedUp && Model.Quest?.IsFinalized == false)
    {
        <div class="mb-3">
            <h6 class="text-muted text-uppercase mb-2">
                <i class="fas fa-vote-yea me-1"></i>Update Your Vote
            </h6>
            <div class="d-grid gap-2">
                <button class="btn btn-success" onclick="changeVoteToYes(@Model.Quest?.Id)">
                    <i class="fas fa-check me-2"></i>Vote Yes
                </button>
                <button class="btn btn-danger" onclick="changeVoteToNo(@Model.Quest?.Id)">
                    <i class="fas fa-times me-2"></i>Vote No
                </button>
                <button class="btn btn-warning" onclick="changeVoteToMaybe(@Model.Quest?.Id)">
                    <i class="fas fa-question me-2"></i>Vote Maybe
                </button>
            </div>
        </div>
    }

    @* Voting section: date selection (not yet signed up, open quest) *@
    @if (User.Identity?.IsAuthenticated == true && !isPlayerSignedUp && Model.Quest?.IsFinalized == false)
    {
        <div class="mb-3">
            <h6 class="text-muted text-uppercase mb-2">
                <i class="fas fa-calendar-alt me-1"></i>Choose a Date
            </h6>
            @foreach (var calendarMonth in ViewBag.CalendarMonths as List<CalendarViewModel> ?? new List<CalendarViewModel>())
            {
                @await Html.PartialAsync("_Calendar", calendarMonth)
            }
        </div>
    }

    @* Participant list (finalized quests) *@
    @if (Model.Quest?.IsFinalized == true && allSelectedParticipants.Count > 0)
    {
        <div class="participant-list-mobile mb-3">
            <h6 class="text-muted text-uppercase mb-2">
                <i class="fas fa-users me-1"></i>Adventurers
            </h6>
            @foreach (var participant in allSelectedParticipants)
            {
                var isCurrentUser = participant.Player.Id == currentUserId;
                string roleBadge = participant.Role == SignupRole.Player ? "bg-primary"
                    : participant.Role == SignupRole.AssistantDM ? "bg-warning text-dark"
                    : "bg-secondary";
                string roleText = participant.Role == SignupRole.Player ? "Player"
                    : participant.Role == SignupRole.AssistantDM ? "Asst. DM"
                    : "Spectator";
                <div class="participant-row d-flex justify-content-between align-items-center py-2 border-bottom @(isCurrentUser ? "bg-dark rounded px-2" : "")">
                    <div>
                        <span class="fw-bold">@participant.Player.Name</span>
                        @if (isCurrentUser)
                        {
                            <span class="badge bg-info ms-1">You</span>
                        }
                        <br>
                        <small class="text-muted">@(participant.Character?.Name ?? "No character")</small>
                    </div>
                    <span class="badge @roleBadge">@roleText</span>
                </div>
            }
        </div>
    }

    @* DM controls link *@
    @if (canManage)
    {
        <a asp-controller="Quest" asp-action="Manage" asp-route-id="@Model.Quest?.Id"
           class="btn btn-warning w-100 mb-2">
            <i class="fas fa-cogs me-2"></i>Manage Quest
        </a>
    }

    @* Revoke signup *@
    @if (User.Identity?.IsAuthenticated == true && isPlayerSignedUp)
    {
        <button type="button" class="btn btn-danger w-100 mt-2"
                onclick="revokeSignup(@Model.Quest?.Id)">
            <i class="fas fa-times me-2"></i>Revoke Signup
        </button>
    }

</div>

@section Scripts {
<script>
    function revokeSignup(questId) {
        if (confirm("Are you sure you want to revoke your signup for this quest? This action cannot be undone.")) {
            const formData = new FormData();
            formData.append('__RequestVerificationToken', '@tokens.RequestToken');
            fetch(`/Quest/RevokeSignup/${questId}`, {
                method: "DELETE",
                body: formData
            }).then(res => {
                if (res.ok) { location.reload(); }
                else { res.text().then(text => { alert(`Failed to revoke signup: ${text}`); }); }
            }).catch(err => { alert("An error occurred while revoking signup."); });
        }
    }

    function changeVoteToYes(questId) {
        if (confirm("Change your vote to Yes and join this quest?")) {
            const formData = new FormData();
            formData.append('__RequestVerificationToken', '@tokens.RequestToken');
            fetch(`/Quest/ChangeVoteToYes/${questId}`, {
                method: "POST",
                body: formData
            }).then(res => {
                if (res.ok) { location.reload(); }
                else { res.text().then(text => { alert(`Failed to change vote: ${text}`); }); }
            }).catch(err => { alert("An error occurred while changing vote."); });
        }
    }

    function changeVoteToNo(questId) {
        if (confirm("Change your vote to No?")) {
            const formData = new FormData();
            formData.append('__RequestVerificationToken', '@tokens.RequestToken');
            fetch(`/Quest/ChangeVoteToNo/${questId}`, {
                method: "POST",
                body: formData
            }).then(res => {
                if (res.ok) { location.reload(); }
                else { res.text().then(text => { alert(`Failed to change vote: ${text}`); }); }
            }).catch(err => { alert("An error occurred while changing vote."); });
        }
    }

    function changeVoteToMaybe(questId) {
        if (confirm("Change your vote to Maybe?")) {
            const formData = new FormData();
            formData.append('__RequestVerificationToken', '@tokens.RequestToken');
            fetch(`/Quest/ChangeVoteToMaybe/${questId}`, {
                method: "POST",
                body: formData
            }).then(res => {
                if (res.ok) { location.reload(); }
                else { res.text().then(text => { alert(`Failed to change vote: ${text}`); }); }
            }).catch(err => { alert("An error occurred while changing vote."); });
        }
    }
</script>
}
```
