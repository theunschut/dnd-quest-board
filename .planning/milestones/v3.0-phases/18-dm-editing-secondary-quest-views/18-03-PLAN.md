**@section Scripts** — copy the file validation JS verbatim from EditProfile.cshtml (per D-13):
```razor
@section Scripts {
    <script>
        const DM_MAX_FILE_SIZE = 5 * 1024 * 1024;
        const DM_ALLOWED_TYPES = ['image/jpeg', 'image/jpg', 'image/png', 'image/gif'];

        document.getElementById('dmProfilePictureInput')?.addEventListener('change', function (e) {
            const file = e.target.files[0];
            const errorDiv = document.getElementById('dmFileSizeError');
            if (file) {
                if (file.size > DM_MAX_FILE_SIZE) {
                    errorDiv.textContent = `File size (${(file.size / 1024 / 1024).toFixed(1)} MB) exceeds the maximum allowed size of 5 MB. Please choose a smaller image.`;
                    errorDiv.style.display = 'block';
                    e.target.value = '';
                    return;
                }
                if (!DM_ALLOWED_TYPES.includes(file.type)) {
                    errorDiv.textContent = 'Only image files (JPG, PNG, GIF) are allowed.';
                    errorDiv.style.display = 'block';
                    e.target.value = '';
                    return;
                }
                errorDiv.style.display = 'none';
                errorDiv.textContent = '';
            }
        });
    </script>
}
```
