    ```csharp
    @functions {
        string BuildTabUrl(ItemType? tabType, IList&lt;ItemRarity&gt; rarities, string? sort)
        {
            var pairs = new List&lt;KeyValuePair&lt;string, string?&gt;&gt;();
            if (tabType.HasValue)
                pairs.Add(new("type", tabType.Value.ToString()));
            foreach (var r in rarities)
                pairs.Add(new("rarity", r.ToString()));
            if (!string.IsNullOrEmpty(sort))
                pairs.Add(new("sort", sort));
            var qs = Microsoft.AspNetCore.Http.QueryString.Create(pairs);
            return Url.Action("Index", "Shop") + qs.Value;
        }
    }
    ```
